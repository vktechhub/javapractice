JPA structure:-
1. EntityManagerFactory emf = Persistence.createEntityManagerFactory("unit-name");
2. EntityManager em = emf.createEntityManager();
3. EntityTransaction tx = em.getTransaction();
4. tx.begin();
5. // Perform CRUD operations
    example: em.persist(entity);    // Create
             em.find(EntityClass.class, id); // Read
             em.merge(entity);               // Update
             em.remove(entity);              // Delete

6. tx.commit();
7. em.close();
8. emf.close();

7. session.close();
8. sessionFactory.close();
